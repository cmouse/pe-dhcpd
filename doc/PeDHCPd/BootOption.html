def initialize(*args)

    @len = args.shift
    @value = args.shift
  elsif args.size == 1
    build args[0]
  elsif args.size == 0
    @key = 0
    @len = 0 
    @value = ""
   else

  end
end

def pack 
  return [@key,@len].pack('CC') + @value if (@value.is_a?(String))
  return ([@key,@len]+@value).pack('CC*') if (@value.is_a?(Array))
  ""
end

def to_s
  str = "Option #{@key} with #{@len} bytes of data: "
  str += @value.unpack("C*").map { |c| "%02x" % [c] }.join(" ")
end
